# Heap

### Min Heap (오름차순)

```Java
import java.util.*;

public class Main {
  public static void main(String[] args) throws Exception {
    PriorityQueue<Integer> pq = new PriorityQueue<>();
    // 최소 힙 (기본)
    // 내부적으로 힙(완전 이진 트리) 구조를 사용하며, O(log N) 시간복잡도를 가진다.

    // 삽입
    pq.offer(3);
    pq.offer(1);
    pq.offer(2);

    pq.peek(); // 조회 (최소값 반환, 제거 X, 비어있으면 null 반환) → 1
    pq.poll(); // 삭제 (최소값 제거 후 반환, 비어있으면 null 반환) → 1

    pq.size();
    pq.isEmpty();

    // 전체 순회 (순서 보장 X, PriorityQueue는 반복문으로 돌린다고 우선순위 순서대로 나오지 않는다.
    // 왜냐하면, 내부적으로 완전정렬 상태를 유지하고 있지 않기 때문이다.
    // 따라서, poll()로 꺼내야 정렬된 순서대로 나온다.)
    for (int x : pq) {
      System.out.println(x);
    }
  }
}
```

### Pair Heap (복합 정렬)

```java
import java.util.*;

class Pair {
  int x, y;

  Pair(int x, int y) {
    this.x = x;
    this.y = y;
  }
}

public class Main {
  public static void main(String[] args) throws Exception {
    // 두 번째 원소(우선 정렬): 오름차순, 첫 번째 원소: 내림차순
    PriorityQueue<Pair> pq = new PriorityQueue<>((a, b) -> {
      if (a.y != b.y)
        return Integer.compare(a.y, b.y);
      else
        return Integer.compare(b.x, a.x);
    });

    pq.offer(new Pair(1, 2));
    pq.offer(new Pair(3, 2));
    pq.offer(new Pair(4, 2));
    pq.offer(new Pair(2, 5));
    pq.offer(new Pair(3, 10));
    pq.offer(new Pair(3, 7));

    while (!pq.isEmpty()) {
      int x = pq.peek().x;
      int y = pq.peek().y;

      pq.poll();

      System.out.println(x + " " + y);
    }

    /*
     * 4 2
     * 3 2
     * 1 2
     * 2 5
     * 3 7
     * 3 10
     */
  }
}
```

### Tuple Heap (복합 정렬)

```java
import java.util.*;

class Tuple {
  int x, y, z;

  Tuple(int x, int y, int z) {
    this.x = x;
    this.y = y;
    this.z = z;
  }
}

public class Main {
  public static void main(String[] args) throws Exception {
    // 세 번째 원소(우선 정렬): 오름차순, 나머지 원소: 내림차순
    PriorityQueue<Tuple> pq = new PriorityQueue<>((a, b) -> {
      if (a.z != b.z)
        return Integer.compare(a.z, b.z);
      else if (a.y != b.y)
        return Integer.compare(b.y, a.y);
      else
        return Integer.compare(b.x, a.x);
    });

    pq.offer(new Tuple(1, 2, 5));
    pq.offer(new Tuple(3, 2, 1));
    pq.offer(new Tuple(3, 10, 8));
    pq.offer(new Tuple(3, 7, 1));
    pq.offer(new Tuple(4, 2, 1));
    pq.offer(new Tuple(2, 5, 3));

    while (!pq.isEmpty()) {
      int x = pq.peek().x;
      int y = pq.peek().y;
      int z = pq.peek().z;

      pq.poll();

      System.out.println(x + " " + y + " " + z);
    }

    /*
     * 3 7 1
     * 4 2 1
     * 3 2 1
     * 2 5 3
     * 1 2 5
     * 3 10 8
     */
  }
}
```
