# Iterator

Set, Map과 같은 컬렉션에서 순회 도중 `set.remove()`, `map.remove()`와 같은 컬렉션 메서드를 통해 특정 원소를 삭제하면,<br>
순회 주체인 `Iterator`의 입장에서는 자신의 제어 밖에서 구조 변경이 일어난 것이므로,<br>
`Iterator`가 관리하는 상태와 실제 컬렉션 상태가 어긋나게 되어 `ConcurrentModificationException`이 발생하게 된다.<br>
따라서 순회 도중 삭제가 필요할 때는 Iterator의 `Iterator.remove()`를 통해 수행해야 한다.

## 사용 예시

### 1. ArrayList

```java
import java.util.*;

public class Main {
  public static void main(String[] args) {
    ArrayList<String> list = new ArrayList<>();
    list.add("apple");
    list.add("banana");
    list.add("cherry");
    list.add("banana");

    Iterator<String> it = list.iterator();

    while (it.hasNext()) {
      String value = it.next(); // 인덱스 순서대로 조회

      if (value.equals("banana")) {
        it.remove(); // Iterator를 통한 삭제
      }
    }

    System.out.println(list); // [apple, cherry]
  }
}
```

### 2. Set

#### 1. 단일 원소

```java
import java.util.*;

public class Main {
  public static void main(String[] args) {
    Set<String> set = new HashSet<>();
    set.add("apple");
    set.add("banana");
    set.add("cherry");

    Iterator<String> it = set.iterator();

    while (it.hasNext()) {
      String value = it.next(); // 조회 순서 보장 X

      if (value.equals("banana")) {
        it.remove();
      }
    }

    System.out.println(set);
  }
}
```

#### 2. Pair 원소

```java
import java.util.*;

class Pair {
  int x, y;

  Pair(int x, int y) {
    this.x = x;
    this.y = y;
  }
}

public class Main {
  public static void main(String[] args) throws Exception {
    Set<Pair> s = new TreeSet<>((a, b) -> {
      if (a.x != b.x)
        return Integer.compare(a.x, b.x);
      else
        return Integer.compare(a.y, b.y);
    });
    // TreeSet은 원소를 정렬하며 저장하므로,
    // Pair와 같은 사용자 정의 객체를 원소로 담으려면 반드시 비교 기준을 제공해야 한다.

    s.add(new Pair(3, 1));
    s.add(new Pair(4, 5));
    s.add(new Pair(5, 2));
    s.add(new Pair(6, 5));
    s.add(new Pair(3, 5));
    s.add(new Pair(1, 7));

    Iterator<Pair> it = s.iterator();
    while (it.hasNext()) {
      Pair p = it.next();

      System.out.println(p.x + " " + p.y);

      if (p.x % 2 == 0)
        it.remove();
    }

    System.out.println("result");

    it = s.iterator();
    while (it.hasNext()) {
      Pair p = it.next();

      System.out.println(p.x + " " + p.y);
    }
  }
}
```

### 3. Map

#### 1. 단일 value

```java
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

public class Main {
  public static void main(String[] args) {
    Map<String, Integer> map = new HashMap<>();
    map.put("apple", 3);
    map.put("banana", 5);
    map.put("cherry", 7);

    Iterator<Map.Entry<String, Integer>> it = map.entrySet().iterator();

    while (it.hasNext()) {
      Map.Entry<String, Integer> entry = it.next(); // 조회 순서 보장 X

      if (entry.getKey().equals("banana")) {
        it.remove();
      }
    }

    System.out.println(map); // { apple=3, cherry=7 }
  }
}
```

#### 2. Pair value

```java
import java.util.*;

class Pair {
  int x, y;

  Pair(int x, int y) {
    this.x = x;
    this.y = y;
  }
}

public class Main {
  public static void main(String[] args) throws Exception {
    Map<Integer, Pair> m = new HashMap<>();

    m.put(1, new Pair(1, 2));
    m.put(2, new Pair(3, 4));
    m.put(3, new Pair(5, 6));
    m.put(4, new Pair(7, 8));
    m.put(5, new Pair(9, 10));

    Iterator<Map.Entry<Integer, Pair>> it = m.entrySet().iterator();
    while (it.hasNext()) {
      Map.Entry<Integer, Pair> e = it.next();
      // 한 반복에서 next()를 여러 번 호출하면 서로 다른 원소를 꺼내게 되므로,
      // 현재 원소 전체를 한 번에 받아 두고 그 안에서 key와 value를 꺼내 써야 한다.

      int key = e.getKey();
      int value_x = e.getValue().x;
      int value_y = e.getValue().y;

      System.out.println(key + " " + value_x + " " + value_y);

      if (key % 2 == 0) {
        it.remove();
      } else {
        e.setValue(new Pair(value_x, value_y * 2));
        // 값 수정
      }
    }

    System.out.println("Result");

    it = m.entrySet().iterator();
    // 기존 iterator는 끝까지 순회하면 재사용할 수 없으므로, 반드시 새 iterator를 만들어야 한다.
    while (it.hasNext()) {
      Map.Entry<Integer, Pair> e = it.next();

      int key = e.getKey();
      int value_x = e.getValue().x;
      int value_y = e.getValue().y;

      System.out.println(key + " " + value_x + " " + value_y);
    }
  }
}
```
